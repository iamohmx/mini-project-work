DESC เรียงจากมากไปหาน้อย
SELECT * FROM customers 
WHERE Country = 'Canada' OR Country ='USA'
ORDER BY CustomerId DESC;

LIMIT จำกัดข้อมูล
SELECT * FROM customers 
WHERE Country = 'Canada' OR Country ='USA'
ORDER BY CustomerId DESC
LIMIT 5;

GROUP BY รวมกลุ่มข้อมูล และนับ
SELECT COUNT(CustomerId),City
FROM customers
GROUP BY City;

GROUP BY รวมกลุ่มข้อมูล และนับและเรียงลำดับข้อมูล
SELECT count(CustomerId) AS CusID,Country
FROM customers
GROUP BY City
ORDER BY CustomerId DESC;

การใช้ Having ใช้กับ Aggregate Function และอยู่หลัง GROUP BY (ถ้ามี)
SELECT count(CustomerId),Country
FROM customers
GROUP BY Country
HAVING count(CustomerId) >=5
ORDER BY CustomerId DESC;
// 2 
SELECT count(CustomerId) AS CusID,Country
FROM customers
GROUP BY Country
HAVING CusID >=5
ORDER BY CusID DESC;

	Column_name(s)
FROM
	table_name

	Condition(s)

	Column_name(s)

	Condition(s)
ORDER BY
	Column_name(s)
LIMIT
	number of rows;

OPERATORS
การใช้ NOT
SELECT * FROM employees
WHERE NOT Title = "IT Staff";

การใช้ IN
SELECT * FROM employees
WHERE Title IN('General Manager','Sales Manager',
'Sales Support Agent')

NOT และ IN
SELECT * FROM customers
WHERE Country NOT IN('Brazil','USA','Sweden')

การใช้ LIKE
SELECT * FROM customers
WHERE
	FirstName LIKE "r%t";
// % คือสามารถเอาตัวษรที่เราอยากค้นหาไปวาง เช่น LIKE "R%B%T" ตัวอักษร R ต้องมีอยู่ข้างหน้า T ต้องตามหลัง B อยู่ตรงไหนก็ได้ภายในชื่อ

การใช้ BETWEEN
SELECT * FROM customers
WHERE
	Country IN ('Brazil','Czech Republic') 
	AND CustomerId BETWEEN 1 AND 30
	
การใช้ IS NULL ดูข้อมูลที่มีช่องว่าง
SELECT * FROM customers
WHERE
	Phone IS NULL; //Aplied Phone IS NOT NULL;
	
การใช้ AND และ  OR
// AND ต้องจริงเท่านั้น, OR จริงหรือเท็จ เป็นจริง
